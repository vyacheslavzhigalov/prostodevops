### git instructions
git init                  // Создать новый локальный репозиторий в текущей папке
git clone <url>           // Клонировать удаленный репозиторий на компьютер
git status                // Показать состояние файлов (измененные, новые, удаленные)
git add <file>            // Добавить конкретный файл в индекс (подготовить к коммиту)
git add .                 // Добавить все измененные и новые файлы в индекс
git commit -m "сообщение" // Зафиксировать изменения с описанием
git commit -am "текст"    // Добавить файлы в индекс и сразу сделать коммит (только для измененных)
git log                  # Показать историю всех коммитов
git log --oneline        # История в компактном виде (одна строка на коммит)
git branch               # Показать список локальных веток
git branch <name>        # Создать новую ветку
git checkout <name>      # Перейти на указанную ветку
git checkout -b <name>   # Создать новую ветку и сразу перейти на нее
git merge <name>         # Слить указанную ветку в текущую
git branch -d <name>     # Удалить ветку (после слияния)
git remote add origin <url> # Связать локальный репозиторий с удаленным
git push origin <branch>    # Отправить изменения в удаленный репозиторий
git pull origin <branch>    # Получить изменения из удаленного репозитория и слить их
git fetch origin            # Загрузить данные из удаленного репозитория, но не объединять
git restore <file>       # Отменить изменения в файле до последнего коммита
git reset --hard HEAD    # Откатить проект до последнего коммита (удалит все изменения)
git commit --amend -m "" # Изменить текст последнего коммита или добавить в него забытые файлы
